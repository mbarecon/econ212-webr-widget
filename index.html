import { WebR, ChannelType } from "https://webr.r-wasm.org/latest/webr.mjs"; async function initWebR() { setStatus("Loading webR..."); webR = new WebR({ channelType: ChannelType.PostMessage, }); await webR.init(); await webR.evalRVoid("options(device=webr::canvas)"); setStatus("webR ready", true); } async function runCode() { const runBtn = document.getElementById("runBtn"); const outputEl = document.getElementById("output"); const plotEl = document.getElementById("plot"); runBtn.disabled = true; setStatus("Running..."); clearOutput(); let shelter = null; try { const userCode = getCode(); shelter = await new webR.Shelter(); const result = await shelter.captureR(userCode, { withAutoprint: true, captureGraphics: { width: 800, height: 500, bg: "white" } }); const lines = []; for (const item of result.output) { if (item.type === "stdout" || item.type === "stderr") { lines.push(item.data); } else if (item.type === "message" || item.type === "warning" || item.type === "error") { lines.push("[" + item.type + "]\n"); if (item.data) lines.push(String(item.data) + "\n"); } } outputEl.textContent = lines.join("") || "Code executed."; if (result.images && result.images.length > 0) { result.images.forEach((img) => { const canvas = document.createElement("canvas"); canvas.width = img.width; canvas.height = img.height; canvas.getContext("2d").drawImage(img, 0, 0, img.width, img.height); plotEl.appendChild(canvas); }); } else { plotEl.textContent = "No plot was generated."; } } catch (err) { outputEl.textContent = "JavaScript error: " + err.message; } finally { if (shelter) shelter.purge(); runBtn.disabled = false; setStatus("webR ready", true); } }
