async function runCode() { runBtn.disabled = true; statusEl.textContent = "Running..."; clearOutput(); let shelter; try { const userCode = codeEl.value; shelter = await new webR.Shelter(); const result = await shelter.captureR(userCode, { captureGraphics: { width: 800, height: 500, bg: "white" } }); // Show text output const lines = []; for (const item of result.output) { if (item.type === "stdout" || item.type === "stderr") { lines.push(item.data); } else if (item.type === "message" || item.type === "warning" || item.type === "error") { lines.push(`[${item.type}]`); } } outputEl.textContent = lines.join(""); // Show plots if (result.images.length > 0) { result.images.forEach((img) => { const canvas = document.createElement("canvas"); canvas.width = img.width; canvas.height = img.height; canvas.style.maxWidth = "100%"; canvas.style.height = "auto"; const ctx = canvas.getContext("2d"); ctx.drawImage(img, 0, 0, img.width, img.height); plotEl.appendChild(canvas); }); } else { plotEl.textContent = "No plot was generated."; } if (!outputEl.textContent.trim() && result.images.length === 0) { outputEl.textContent = "Code executed."; } } catch (err) { outputEl.textContent = "JavaScript error: " + err.message; } finally { if (shelter) { shelter.purge(); } runBtn.disabled = false; statusEl.textContent = "webR ready"; } }
